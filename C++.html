១/ ការណែនាំអំពី C++


C++ គឺជាភាសាកុំព្យូរទ័រដែលត្រូវបានបង្កើតឡើងដោយលោក Bjame Stroustrup។ ដំបូងឡើយភាសា C++ គេហៅថា C with Classes ប៉ុន្តែក្រោយមកទើបប្តូរមកជា C++។C++ បានចេញផ្សាយមុនដំបូងគេបង្អស់នៅឆ្នាំ ១៩៨៣។

#include <iostream>: គឺជា header file សម្រាប់ប្រើនៅក្នុងភាសា C++។
using namespace std;: មានន័យថាយើងបានប្រើប្រាស់ namespace ដែលមានឈ្នោះថា std។
Statement: គឺជាតំបន់សម្រាប់សរសេរ Statement ដែលយើងចង់សរសេរ។


សម្គាល់៖​ ដើម្បីធ្វើការ Output តម្លៃណាមួយ ឬទិន្នន័យណាមួយមកលើផ្ទាំង Console បានចំពោះភាសា C++ គឺយើងត្រូវប្រើ cout រួមជាមួយនឹងសញ្ញា <<។

	return 0;

២/ ការប្រើប្រាស់ Comment C++


Comment៖អាចត្រូវបានប្រើដើម្បីពន្យល់ឬកត់ត្រានៅត្រង់កន្លែងណាមួយដែលProgrammerគិតថាសំខាន់។យើងអាចCommentម្ដងតែមួយជួរ(singled-line) ក៏បានឬក៏Commentជាលក្ខណះច្រើនជួរក៏បាន(multi-line)។


២.១/ ការប្រើប្រាស់ Single-line


Single-line៖ចាប់ផ្ដើមដោយសញ្ញាforward-slashes(//)។រាល់Statementដែលប្រើសញ្ញា//នៅពីមុខគឺមិនត្រូវបានត្រួតពិនិត្យដោយCompilerឡើយ។


#ឧទាហរណ៍៖ ការប្រើប្រាស់ singled-line Comment 
					
	#include <iostream>
	using namespace std;
	int main() {
		cout << "Hello World!";// This is a comment
	
	return 0;
	}
#ចម្លើយ៖​ 
Hello, World!

២.២/ ការប្រើប្រាស់ multi-line Comments


Multi-line Comments៖ចាប់ផ្ដើមដោយសញ្ញា/*។និងបញ្ចប់ដោយសញ្ញា*/ដែលប្រើសញ្ញា//។រាល់StatementឬTextដែលនៅចន្លោះសញ្ញាទាំងពីរនេះគឺមិនត្រូវបានដំណើរការឬក៏ត្រួតពិនិត្យដោយCompilerឡើយ។


#ឧទាហរណ៍៖ ការប្រើប្រាស់ Multi-line Comment 
					
	#include <iostream>
	using namespace std;
	int main() {
		/* The code below will print the words Hello world! to the screen, and it is amazing */
		cout << "Hello World!";
	
	return 0;
	}

Hello, World!

៣/ ការប្រើប្រាស់ C++ Variable


Variable៖ប្រៀបបានជារបស់ម៉្យាង(ប្រអប់ឬcontainer)ដែលយើងប្រើវាសម្រាប់ផ្ទុកទិន្នន័យផ្សេងៗ។ជាក់ស្ដែងនៅក្នុងC++មានVariableជាច្រើនប្រភេទដែលប្រភេទនីមួយៗត្រូវបានកំណត់ដោយDataType។ដូច្នេះVariableមួយត្រូវតែនៅអមដោយDataTypeណាមួយជានិច្ច។


#ឧទាហរណ៍៖ DataType មួយចំនួនដែលគេប្រើជាញឹកញាប់នៅក្នុងភាសា C++

int : អាចឲ variable ផ្ទុកនូវចំនួនគត់ (integer)។ Ex:-1, 2, -12, 100...
double : អាចឲ​ variable ផ្ទុកនូវចំនួនទសភាគ (floating point number)។​ Ex: 1.2, 12.33, 11.00,...
int : អាចឲ variable ផ្ទុកនូវចំនួនគត់ (integer)។ Ex:-1, 2, -12, 100...
char : អាចឲ​ variable : ផ្ទុកនូវចំនួនតួអក្សរតែមួយតួរគត់ (single character)។ Ex:'a', 'A', 'B',...។ជាទូទៅតម្លៃរបស់វាត្រូវបានសរសេរដាក់នៅក្នុងសញ្ញា Double quote(' ')
string : អាចឲ variable ផ្ទុកនូវពាក្យឬឃ្លាជាច្រើន។ Ex:"Hello World", "How are you?",...។​ តម្លៃរបស់វាត្រូវបានគេសរសេរដាក់នៅក្នុងសញ្ញា Double quote(" ")
bool : អាចឲ variable ផ្ទុកនូវតម្លៃតែពីរគត់គឺ ពិត(True) ឬមិនពិត(False)។



៣.១/ ការប្រកាស (បង្កើត) Variable


#Syntax:
DataType Variable_Name = value;

DataType :int, char, double, string, bool,...
Variable_Name :ជាឈ្មោះរបស់ variable។ Ex: myName,x,y,...
Value :ជាតម្លៃដែលត្រូវបានកំណត់អាចជាលេខទសភាគ,តួរអក្សរ ឬចំនួនគត់ ឬផ្សេងៗទៀតទៅតាមប្រភេទទិន្នន័យ(DataType)
= :ជា Assignment Operator ប្រើសម្រាប់ផ្ទេរតម្លៃរបស់​ value ទៅឲ variable ដែលនៅខាងស្ដាំសញ្ញារបស់វា។

៥/ C++ User Input


  យើងបានសិក្សារួចមកហើយអំពី keyword cout ដែលយើងប្រើវាសម្រាប់ធ្វើការ Output(print) តម្លៃណាមួយមកលើផ្ទាំង Console។ ឥឡូវយើងនឹងរៀនអំពី​ keyword មួយទៀតគឺ cin ដែលយើងប្រើវាក្នុងគោលបំណងដើម្បីអាចអោយយើងធ្វើការ input។
cinត្រូវបានគេប្រើសម្រាប់អាន(read) ទិន្នន័យដែលត្រូវបានបញ្ចូលតាមរយះKeyboard ហើយKeyboardនេះត្រូវបានគេប្រើជាមួយនឹងសញ្ញា Extraction Operator(>>)។

    Increment(++):វានឹងធ្វើអោយតម្លៃដើមកើនចំនួន១
    Decrement(--):វានឹងធ្វើអោយតម្លៃដើមថយចំនួន១។

#ឧទាហរណ៍៖ បន្ថែមអំពីការប្រើប្រាស់ Increment Operator និង Decrement Operator 
					
#include <iostream>
using namespace std;
int main() {
	int a = 10, b = 10, result_a, result_b;
	// យើងអោយ a កើនតម្លៃចំនួន 1 រួចហើយផ្ទុកតម្លៃនោះនៅក្នុង rwsult_a
	result_a = ++a;
	cout << "Result_a = " << result_a << endl;
	// យើងអោយ b ថយតម្លៃចំនួន 1 រួចហើយផ្ទុកតម្លៃនោះនៅក្នុង result_b
	result_b = --b;
	cout << "Result_b = " << result_b << endl;

    return 0;
}

	cout << "(3 == 5) && (3 > 5) is " << result << endl;

	cout << "(3 != 5) || (3 < 5) is " << result << endl;
